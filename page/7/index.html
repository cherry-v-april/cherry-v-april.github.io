手动添加修改信息

手动添加修改信息

```php

 public function edit()
{
    $dbConfig = config("db_fenxiao_config");
    $request = Request::instance();
    if ($request->isPost()) {
        $addressData = input('post.');
        if (isset($addressData['id'])) {
            $expressModel = Db::connect($dbConfig)->table("fx_express");
            $result = $expressModel->update($addressData);
            if ($result !== false) {
                $this->success('恭喜，保存成功哦！', '');
            } else {
                $this->error('保存失败，请稍候再试！');
            }
        }
    } elseif ($request->isGet()) {
        $id = $_GET['id'];

        $expressModel = Db::connect($dbConfig)->table("fx_express");
        $vo = $expressModel->where('id', $id)->find();

        if (isset($vo)) {
            $this->assign('vo', $vo);
        }
        $this->assign('id', $id);
        return $this->fetch('form');
    } else {
        return null;
    }
    return null;


}
